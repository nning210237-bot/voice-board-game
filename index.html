        state.teams.push({

            id: i, name: document.getElementById(`name-${i}`).value, ml: 10,

            cards: shuffledCards.slice(i*4, (i+1)*4).map(c => ({...c, used: false})),

            isEliminated: false, hintCardUsed: false, misledBy: null

        });

    }

    document.getElementById('setup-modal').classList.add('hidden');

    makeDraggable(document.getElementById("panel-left"));

    makeDraggable(document.getElementById("panel-main"));

 

    if (!state.timerStarted) {

        state.timerStarted = true;

        setInterval(() => {

            if(state.gameTime > 0) {

                state.gameTime--;

                const m = Math.floor(state.gameTime / 60);

                const s = state.gameTime % 60;

                document.getElementById('game-timer').innerText = `${m}:${s < 10 ? '0' : ''}${s}`;

            }

        }, 1000);

    }

    updateUI();

    logBattle(`🍄 玩家 [${state.userRole === 'admin' ? '管理員' : state.teams[state.userRole].name}] 已就緒！`);

}

 

function updateUI() {

    const scoreboard = document.getElementById('global-scoreboard');

    if (!scoreboard) return;

    scoreboard.innerHTML = state.teams.map((t, i) => {

        let activeClass = '';

        if (state.isCoopMode) {

            if (i === state.activeTurnIdx || i === state.cardTargetIdx) activeClass = 'coop-glow scale-105';

        } else if (i === state.actingTeamIdx) {

            activeClass = 'ring-4 ring-morandi-red scale-105';

        }

        return `

            <div class="flex-1 bg-white p-4 border-2 border-morandi-dark ${t.isEliminated ? 'opacity-30' : activeClass} transition-all rounded-2xl shadow-sm">

                <div class="flex justify-between items-center mb-1 text-slate-800">

                    <span class="text-2xl">${i===0?'🟦':i===1?'🟥':'🟩'}</span>

                    <span class="font-black text-2xl">${t.ml} 毫升</span>

                </div>

                <div class="text-[9px] font-bold truncate text-slate-400 uppercase">${t.name}</div>

                <div class="pipe-container mt-1"><div class="pipe-fill" style="width: ${Math.min(100, t.ml * 10)}%"></div></div>

            </div>

        `;

    }).join('');

 

    const overlay = document.getElementById('spectator-overlay');

    if (overlay) {

        const isMyTurn = state.userRole === state.activeTurnIdx;

        const isAdmin = state.userRole === 'admin';

        overlay.classList.toggle('hidden', isMyTurn || isAdmin || state.isStealing || state.isCoopMode);

    }

 

    const cardGridEl = document.getElementById('card-grid');

    if (cardGridEl && state.userRole !== 'admin') {

        const myTeam = state.teams[state.userRole];

        cardGridEl.innerHTML = myTeam.cards.map((c, i) => `

            <div onclick="handleCardClick(${i})" class="chance-card p-3 flex flex-col gap-1 ${c.used || (state.cardUsedThisTurn && state.activeTurnIdx === state.userRole) ? 'disabled' : ''}">

                <div class="flex items-center gap-2 text-morandi-dark">

                    <span class="material-symbols-outlined text-xl">${c.icon}</span>

                    <span class="font-black text-sm uppercase">${c.name}</span>

                </div>

                <p class="text-[9px] font-bold text-slate-400 leading-tight">${c.rules}</p>

            </div>

        `).join('');

    } else if (cardGridEl && state.userRole === 'admin') {

        cardGridEl.innerHTML = `<p class="text-center text-slate-400 italic py-10">講師主控模式：監看全局狀態</p>`;

    }

 

    if (document.getElementById('current-team-name')) {

        document.getElementById('current-team-name').innerText = state.isStealing ? "⚡ 搶答中！" : state.teams[state.activeTurnIdx].name;

    }

    if (document.getElementById('pool-count')) {

        document.getElementById('pool-count').innerText = `剩餘：${state.taskPool.length} 題`;

    }

    const hintBtn = document.getElementById('hint-btn');

    if(hintBtn) hintBtn.disabled = (state.currentTask === null || state.teams[state.activeTurnIdx].hintCardUsed);

}

 

function handleCardClick(idx) {

    if(state.cardUsedThisTurn) return;

    if(state.userRole !== state.activeTurnIdx && state.userRole !== 'admin') return;

 

    const team = state.teams[state.activeTurnIdx];

    const card = team.cards[idx];

   

    if(card.id === 10) {

        const enemies = state.teams.filter((t, i) => i !== state.activeTurnIdx && !t.isEliminated);

        if(enemies.length > 0) {

            useCard(idx, enemies[Math.floor(Math.random() * enemies.length)].id);

        }

        return;

    }

 

    const needsTarget = [2, 4, 5, 9, 12, 13].includes(card.id);

    if(needsTarget) {

        showTargetSelection(idx);

    } else {

        openConfirmModal(idx, null);

    }

}

 

function showTargetSelection(cardIdx) {

    const modal = document.createElement('div');

    modal.className = 'fixed inset-0 z-[9600] modal-overlay flex items-center justify-center p-4';

    modal.innerHTML = `

        <div class="bg-white border-4 border-morandi-dark p-8 max-w-md w-full text-center rounded-[32px] shadow-2xl">

            <h2 class="text-2xl font-black mb-6 italic uppercase">選擇施放對象</h2>

            <div class="space-y-3">

                ${state.teams.map((t, i) => {

                    if(i === state.activeTurnIdx || t.isEliminated) return '';

                    return `<button onclick="this.parentElement.parentElement.parentElement.remove(); openConfirmModal(${cardIdx}, ${i})" class="w-full py-4 bg-morandi-green text-white mario-btn font-black text-xl">

                        ${i===0?'🟦':i===1?'🟥':'🟩'} ${t.name}

                    </button>`;

                }).join('')}

            </div>

            <button onclick="this.parentElement.parentElement.remove()" class="mt-6 font-black text-slate-400 uppercase underline">取消</button>

        </div>

    `;

    document.body.appendChild(modal);

}

 

function openConfirmModal(cardIdx, targetIdx) {

    const card = state.teams[state.activeTurnIdx].cards[cardIdx];

    document.getElementById('confirm-card-name').innerText = card.name;

    document.getElementById('confirm-card-desc').innerText = card.rules;

    document.getElementById('btn-confirm-use').onclick = function() {

        useCard(cardIdx, targetIdx);

        closeConfirmModal();

    };

    document.getElementById('confirm-modal').style.display = 'flex';

}

 

function closeConfirmModal() { document.getElementById('confirm-modal').style.display = 'none'; }

 

function useCard(cardIdx, targetIdx) {

    const team = state.teams[state.activeTurnIdx];

    const card = team.cards[cardIdx];

    card.used = true;

    state.cardUsedThisTurn = true;

    state.cardActive = card.id;

    state.cardTargetIdx = targetIdx;

   

    if(card.id === 10) {

        state.teams[targetIdx].misledBy = state.activeTurnIdx;

        logBattle(`✨ 心理誤導已埋設至某組。`);

    } else {

        logBattle(`✨【${team.name}】使用：${card.name}`);

    }

 

    if(card.id === 1) { clearInterval(state.timerInt); document.getElementById('ans-timer').innerText = "∞"; }

    if(card.id === 2) { state.isCoopMode = true; state.actingTeamIdx = null; }

    if(card.id === 3) { state.doubleChallenge = true; state.doubleCount = 0; }

    if(card.id === 7) state.riskMode = true;

    if(card.id === 8) state.maskProtected = true;

    if([4, 5, 9, 12].includes(card.id)) { state.actingTeamIdx = targetIdx; state.isSpectating = true; }

    updateUI();

}

 

function drawTask() {

    if (state.userRole !== state.activeTurnIdx && state.userRole !== 'admin') return;

    if (state.taskPool.length === 0) return;

    const randomIdx = Math.floor(Math.random() * state.taskPool.length);

    const task = state.taskPool.splice(randomIdx, 1)[0];

    state.currentTask = task;

    if (state.actingTeamIdx === null && !state.isCoopMode) state.actingTeamIdx = state.activeTurnIdx;

   

    speak(task.q);

    renderTaskBody(task);

    startAnsTimer();

    updateUI();

}

 

function renderTaskBody(task) {

    let taskHTML = `<div class="bg-white border-2 border-morandi-dark p-10 w-full max-w-3xl rounded-[32px] shadow-sm">

            <span class="bg-exam-purple text-white px-6 py-2 border-2 border-morandi-dark font-black mb-6 inline-block italic rounded-full">

                ${task.type}

            </span>

            <h2 class="text-3xl font-black leading-tight mb-10 text-morandi-dark">${task.q}</h2>`;

 

    if(task.opts) {

        taskHTML += `<div class="grid grid-cols-1 gap-4">`;

        task.opts.forEach((opt, i) => {

            taskHTML += `<button onclick="confirmTask(${i === task.a})" class="p-6 bg-white border-2 border-morandi-dark hover:bg-morandi-yellow font-black text-xl mario-btn transition-all text-left flex gap-4">

                <span class="opacity-30">選項 ${String.fromCharCode(65+i)}</span> ${opt}</button>`;

        });

        taskHTML += `</div>`;

    } else {

        taskHTML += `<div class="flex gap-6">

            <button onclick="confirmTask(true)" class="flex-1 py-10 bg-morandi-green text-white mario-btn font-black text-4xl">正確</button>

            <button onclick="confirmTask(false)" class="flex-1 py-10 bg-morandi-red text-white mario-btn font-black text-4xl">錯誤</button>

        </div>`;

    }

    taskHTML += `</div>`;

    document.getElementById('mission-body').innerHTML = taskHTML;

}

 

function confirmTask(isSuccess) {

    clearInterval(state.timerInt);

    if(isSuccess) {

        if(state.doubleChallenge) {

            state.doubleCount++;

            if(state.doubleCount < 2) { logBattle("🦾 第一關成功！續抽下一題。"); setTimeout(drawTask, 1000); return; }

            state.pendingExtraRolls = 1; state.doubleChallenge = false;

        }

        state.scoringTeams = (state.cardActive === 4) ? [state.activeTurnIdx] : (state.isCoopMode || state.cardActive === 5) ? [state.activeTurnIdx, state.cardTargetIdx] : [state.actingTeamIdx || state.activeTurnIdx];

        openDiceModal('plus');

    } else {

        state.scoringTeams = (state.cardActive === 5 || state.cardActive === 4) ? [state.cardTargetIdx] : (state.isCoopMode) ? [state.activeTurnIdx, state.cardTargetIdx] : [state.actingTeamIdx || state.activeTurnIdx];

        state.mustStealAfterDice = !state.isStealing && !state.hasBeenStolen;

        openDiceModal('minus');

    }

}

 

function openDiceModal(mode) {

    state.diceMode = mode;

    document.getElementById('dice-modal').style.display = 'flex';

    document.getElementById('dice-cube').innerText = "?";

    document.getElementById('roll-btn').disabled = false;

    document.getElementById('roll-status-msg').innerText = mode==='plus'?'🌟 獎勵環節':'💀 懲罰環節';

}

 

function executeRoll() {

    const cube = document.getElementById('dice-cube');

    const rollBtn = document.getElementById('roll-btn');

    rollBtn.disabled = true;

    cube.classList.add('dice-rolling');

   

    setTimeout(() => {

        let val = Math.floor(Math.random() * 6) + 1;

        cube.innerText = val;

        cube.classList.remove('dice-rolling');

        let mult = state.riskMode ? 2 : 1;

        if(state.diceMode === 'minus' && state.maskProtected) { val = 0; mult = 1; logBattle("🛡️ 口罩生效！免傷。"); state.maskProtected = false; }

 

        const change = (state.diceMode === 'plus' ? val : -val) * mult;

        state.scoringTeams.forEach(tid => {

            state.teams[tid].ml = Math.max(0, state.teams[tid].ml + change);

            if(state.teams[tid].ml <= 0) state.teams[tid].isEliminated = true;

        });

        updateUI();

 

        const survivors = state.teams.filter(t => !t.isEliminated);

        if (survivors.length <= 1) {

            setTimeout(() => { document.getElementById('dice-modal').style.display = 'none'; endGame(survivors[0]?.name || '和局'); }, 1500);

            return;

        }

       

        setTimeout(() => {

            if((state.cardActive === 6 || state.pendingExtraRolls > 0) && state.diceMode === 'plus') {

                if (state.pendingExtraRolls > 0) state.pendingExtraRolls--; else state.cardActive = null;

                cube.innerText = "?"; rollBtn.disabled = false; return;

            }

            document.getElementById('dice-modal').style.display = 'none';

            if(state.mustStealAfterDice) { state.mustStealAfterDice = false; startStealCountdown(); }

            else { finishTurn(); }

        }, 1500);

    }, 1200);

}

 

function startStealCountdown() {

    document.getElementById('steal-overlay').classList.remove('hidden');

    let sTime = 5;

    const int = setInterval(() => {

        sTime--; document.getElementById('steal-countdown').innerText = sTime;

        if(sTime <= 0) { clearInterval(int); document.getElementById('steal-overlay').classList.add('hidden'); finishTurn(); }

    }, 1000);

 

    document.getElementById('steal-buttons').innerHTML = state.teams.map((t, i) => {

        if(i === state.actingTeamIdx || t.isEliminated) return '';

        return `<button onclick="triggerSteal(${i}, ${int})" class="px-10 py-6 bg-morandi-red text-white mario-btn font-black text-2xl shadow-xl">我是 ${t.name}！我來搶！</button>`;

    }).join('');

}

 

function triggerSteal(idx, intervalId) {

    clearInterval(intervalId);

    state.isStealing = true; state.hasBeenStolen = true; state.actingTeamIdx = idx;

    document.getElementById('steal-overlay').classList.add('hidden');

    state.isCoopMode = false; state.ansTimer = 60;

    logBattle(`⚡【${state.teams[idx].name}】搶到了答題權！`);

    startAnsTimer(); updateUI();

}

 

function startAnsTimer() {

    clearInterval(state.timerInt);

    state.timerInt = setInterval(() => {

        if(state.ansTimer > 0) {

            state.ansTimer--;

            document.getElementById('ans-timer').innerText = state.ansTimer;

        } else { confirmTask(false); }

    }, 1000);

}

    state.isSpectating = false; state.isStealing = false; state.hasBeenStolen = false; state.isCoopMode = false;

    state.ansTimer = 90; state.cardActive = null; state.currentTask = null;

    state.riskMode = false; state.maskProtected = false; state.cardUsedThisTurn = false;

    updateUI();

}

function makeDraggable(el) {

    let pos1 = 0, pos2 = 0, pos3 = 0, pos4 = 0;

    const handle = el.querySelector(".drag-handle");

    if (handle) handle.onmousedown = dragMouseDown;

    function dragMouseDown(e) {

        e.preventDefault(); pos3 = e.clientX; pos4 = e.clientY;

        document.onmouseup = closeDragElement; document.onmousemove = elementDrag;

        el.style.zIndex = 1000;

        document.querySelectorAll('.mario-block').forEach(p => { if(p !== el) p.style.zIndex = 40; });

    }

    function elementDrag(e) {

        e.preventDefault(); pos1 = pos3 - e.clientX; pos2 = pos4 - e.clientY;

        pos3 = e.clientX; pos4 = e.clientY;

        el.style.top = (el.offsetTop - pos2) + "px"; el.style.left = (el.offsetLeft - pos1) + "px";

    }

    function closeDragElement() { document.onmouseup = null; document.onmousemove = null; }

}
